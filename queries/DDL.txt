CREATE DATABASE Futebol

USE Futebol

CREATE TABLE Times (
	Id INT NOT NULL IDENTITY(1,1) PRIMARY KEY,
	Nome VARCHAR(100)
)

CREATE TABLE Jogos (
	Id INT NOT NULL IDENTITY(1,1) PRIMARY KEY,
	Time1 INT NOT NULL,
	Time2 INT NOT NULL,
	FOREIGN KEY (Time1) REFERENCES Times(Id),
	FOREIGN KEY (Time2) REFERENCES Times(Id)
)
